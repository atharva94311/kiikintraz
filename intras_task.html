# returns
data['return_1d'] = data.groupby('Coin')['Close'].pct_change()

# short-term momentum
data['momentum_2'] = data.groupby('Coin')['Close'].pct_change(2)
data['momentum_3'] = data.groupby('Coin')['Close'].pct_change(3)

# moving averages
data['ma_7'] = data.groupby('Coin')['Close'].transform(lambda x: x.rolling(7).mean())
data['ma_14'] = data.groupby('Coin')['Close'].transform(lambda x: x.rolling(14).mean())

# ratios (VERY IMPORTANT)
data['ma_ratio'] = data['ma_7'] / data['ma_14']
data['price_vs_ma7'] = data['Close'] / data['ma_7']

# volatility
data['volatility_7'] = data.groupby('Coin')['return_1d'].transform(lambda x: x.rolling(7).std())

# acceleration
data['acceleration'] = data['return_1d'] - data.groupby('Coin')['return_1d'].shift(1)

# volume spike
data['vol_spike'] = data['Volume'] / data.groupby('Coin')['Volume'].transform(lambda x: x.rolling(7).mean())

# Grouped Volatility (Rolling 30-day)
data['vol_30'] = data.groupby('Coin')['Close'].transform(lambda x: x.rolling(30).std())
data['vol_relative'] = data['volatility_7'] / data['vol_30']

# Grouped Z-Score (Distance from its own 7-day mean)
# Note: Ensure 'ma_7' and 'volatility_7' were also calculated using groupby earlier
data['z_score_7'] = (data['Close'] - data['ma_7']) / data['volatility_7']

# Grouped RSI Calculation (14-day)
def calculate_rsi(series, period=14):
    delta = series.diff()
    gain = delta.clip(lower=0)
    loss = -delta.clip(upper=0)
    avg_gain = gain.rolling(period).mean()
    avg_loss = loss.rolling(period).mean()
    rs = avg_gain / avg_loss
    return 100 - (100 / (1 + rs))

data['RSI_14'] = data.groupby('Coin')['Close'].transform(calculate_rsi)
# Drop NaNs (removes 30-day warmup at start and shifted row at end for EVERY coin)
data.dropna(inplace=True)

#using hyerparameter tuning to find the best arameters for my model
from sklearn.model_selection import GridSearchCV, PredefinedSplit
from sklearn.ensemble import RandomForestClassifier
import numpy as np

# 1. Combine Train and Val for GridSearch
X_grid = pd.concat([X_train, X_val])
y_grid = pd.concat([y_train, y_val])

# 2. Create a marker: -1 for training rows, 0 for validation rows
split_index = np.concatenate([
    -1 * np.ones(len(X_train)),
     0 * np.ones(len(X_val))
])
pds = PredefinedSplit(test_fold=split_index)

# 3. Define Parameter Grid
param_grid = {
    'n_estimators': [100, 200],
    'max_depth': [6, 10, 15],
    'min_samples_leaf': [5, 10],
    'class_weight': ['balanced', None] # 'balanced' helps with the 0.002 threshold
}

# 4. Run GridSearch
rf = RandomForestClassifier(random_state=42)
grid_search = GridSearchCV(estimator=rf, param_grid=param_grid, cv=pds, scoring='precision')
grid_search.fit(X_grid, y_grid)

print(f"Best Params: {grid_search.best_params_}")
